通过JavaScript判断浏览器UA实现自动跳转

  2. <script>
  3.         // 获取浏览器的 User-Agent 字符串
  4.         const userAgent = navigator.userAgent.toLowerCase();

  6.         // 定义不同操作系统对应的跳转网址
  7.         const redirectUrls = {
  8.             windows: 'https://www.shireyishunjian.com/main/portal.php?mod=index&mobile=no',
  9.             macos: 'https://www.shireyishunjian.com/main/portal.php?mod=index&mobile=no',
  10.             ios: 'https://www.shireyishunjian.com/main/misc.php?mod=mobile',
  11.             android: 'https://www.shireyishunjian.com/main/misc.php?mod=mobile'
  12.         };

  14.         // 判断操作系统类型并跳转
  15.         if (userAgent.indexOf('win') !== -1) {
  16.             // Windows 系统
  17.             window.location.href = redirectUrls.windows;
  18.         } else if (userAgent.indexOf('mac') !== -1) {
  19.             // macOS 系统
  20.             window.location.href = redirectUrls.macos;
  21.         } else if (userAgent.indexOf('iphone') !== -1 || userAgent.indexOf('ipad') !== -1 || userAgent.indexOf('ipod') !== -1) {
  22.             // iOS 系统
  23.             window.location.href = redirectUrls.ios;
  24.         } else if (userAgent.indexOf('android') !== -1) {
  25.             // Android 系统
  26.             window.location.href = redirectUrls.android;
  27.         } else {
  28.             // 其他系统
  29.             console.log('未识别的操作系统，不进行跳转');
  30.         }
  31. </script>
